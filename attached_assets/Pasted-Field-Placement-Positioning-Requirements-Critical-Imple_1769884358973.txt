3. Draggable Field Implementation
Use react-draggable or similar library with constrained bounds:
bashnpm install react-draggable react-resizable
jsximport Draggable from 'react-draggable';
import { Resizable } from 'react-resizable';

const DraggableResizableField = ({ field, onUpdate }) => {
  const [position, setPosition] = useState({ x: field.x_position, y: field.y_position });
  const [size, setSize] = useState({ width: field.width, height: field.height });

  const handleDrag = (e, data) => {
    const newPosition = { x: data.x, y: data.y };
    setPosition(newPosition);
  };

  const handleDragStop = (e, data) => {
    // Save to database when drag ends
    onUpdate(field.id, {
      x_position: data.x,
      y_position: data.y,
      x_percent: data.x / parentWidth,
      y_percent: data.y / parentHeight
    });
  };

  const handleResize = (e, { size }) => {
    setSize({ width: size.width, height: size.height });
  };

  const handleResizeStop = (e, { size }) => {
    // Save to database when resize ends
    onUpdate(field.id, {
      width: size.width,
      height: size.height
    });
  };

  return (
    <Draggable
      position={position}
      onDrag={handleDrag}
      onStop={handleDragStop}
      bounds="parent"  // CRITICAL: Keeps field within PDF bounds
      grid={[1, 1]}    // Optional: snap to pixel grid for precision
    >
      <Resizable
        width={size.width}
        height={size.height}
        onResize={handleResize}
        onResizeStop={handleResizeStop}
        minConstraints={[50, 20]}  // Minimum field size
        maxConstraints={[400, 100]} // Maximum field size
      >
        <div className={`field-box field-${field.field_type}`} style={{
          width: size.width,
          height: size.height,
          border: `2px solid ${field.signer_color}`,
          backgroundColor: `${field.signer_color}20`,
          cursor: 'move',
          position: 'absolute',
          userSelect: 'none'
        }}>
          <div className="field-label">{field.field_type}</div>
          {/* Resize handles in corners */}
          <div className="resize-handle" />
        </div>
      </Resizable>
    </Draggable>
  );
};

5. Field Snapping & Alignment (Optional but Recommended)
Add snap-to-grid or snap-to-other-fields:
javascriptconst SNAP_THRESHOLD = 5; // pixels

const handleDrag = (e, data) => {
  let x = data.x;
  let y = data.y;
  
  // Snap to other fields
  fields.forEach(otherField => {
    if (Math.abs(x - otherField.x_position) < SNAP_THRESHOLD) {
      x = otherField.x_position;
    }
    if (Math.abs(y - otherField.y_position) < SNAP_THRESHOLD) {
      y = otherField.y_position;
    }
  });
  
  setPosition({ x, y });
};

  const pdfPageDimensions = getPDFPageSize(pageNumber);
  
  await db.query(`
    UPDATE signature_fields 
    SET 
      x_position = $1,
      y_position = $2,
      width = $3,
      height = $4,
      x_percent = $5,
      y_percent = $6,
      width_percent = $7,
      height_percent = $8
    WHERE id = $9

    position.x,
    position.y,
    position.width,
    position.height,
    position.x / pdfPageDimensions.width,
    position.y / pdfPageDimensions.height,

    position.height / pdfPageDimensions.height,
    fieldId
  ]);
};

8. Prevent Field Drift Issues
Key points to prevent fields from moving unexpectedly:

Use bounds="parent" on Draggable to constrain movement
Set position="absolute" on field container
Store field state in component state AND database
Use controlled components (value props, not defaultValue)
Debounce database saves (save on dragStop, not during drag)
Use transform CSS for positioning during drag (more performant than top/left)
Ensure PDF container has position: relative
Lock fields when document is sent for signing (add disabled prop)
