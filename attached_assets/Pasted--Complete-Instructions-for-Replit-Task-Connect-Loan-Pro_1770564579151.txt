### STEP 4: Auto-Populate Deal Pipeline on Deal Creation
**Location:** Deal creation/conversion logic (when quote becomes a deal, or when deal is first created)

**Core Logic to Add:**
```javascript
// When a new deal is created:
async function createDealPipeline(dealId, programId) {
  // 1. Fetch program template
  const program = await db.query(`
    SELECT * FROM loan_programs WHERE id = ?
  `, [programId]);
  
  if (!program) {
    console.error('Program not found:', programId);
    return; // Exit gracefully if no program
  }
  
  // 2. Fetch stages for this program
  const stages = await db.query(`
    SELECT * FROM program_stages 
    WHERE program_id = ? 
    ORDER BY stage_order ASC
  `, [programId]);
  
  // 3. For each stage, create deal stages
  for (const stage of stages) {
    const dealStageId = await db.insert('deal_stages', {
      deal_id: dealId,
      stage_name: stage.stage_name,
      stage_order: stage.stage_order,
      status: 'pending',
      created_at: new Date()
    });
    
    // 4. Fetch and create tasks for this stage
    const tasks = await db.query(`
      SELECT * FROM program_tasks 
      WHERE stage_id = ? 
      ORDER BY task_order ASC
    `, [stage.id]);
    
    for (const task of tasks) {
      await db.insert('deal_tasks', {
        deal_stage_id: dealStageId,
        task_name: task.task_name,
        is_required: task.is_required,
        status: 'pending',
        created_at: new Date()
      });
    }
  }
  
  // 5. Fetch and create document requirements
  const documents = await db.query(`
    SELECT * FROM program_documents 
    WHERE program_id = ?
  `, [programId]);
  
  for (const doc of documents) {
    await db.insert('deal_documents', {
      deal_id: dealId,
      document_name: doc.document_name,
      is_required: doc.is_required,
      stage_id: doc.stage_id,
      status: 'pending',
      created_at: new Date()
    });
  }
  
  console.log(`Deal pipeline created for deal ${dealId} using program ${programId}`);
}

// Call this function immediately after a deal is created:
// Example:
const newDealId = await createDeal(quoteData);
await createDealPipeline(newDealId, quoteData.program_id);
```
