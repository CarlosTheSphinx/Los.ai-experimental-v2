4️⃣ Example: seed 20 fake borrower users

This example assumes:

You have a User model

You store role + application status

You hash passwords normally

// server/scripts/seedBorrowers.js
import { faker } from "@faker-js/faker";
import bcrypt from "bcrypt";
import db from "../db"; // adjust to your db import

const BORROWER_STAGES = [
  "registered",
  "started_application",
  "submitted",
  "under_review",
  "approved",
  "declined",
  "funded",
  "needs_docs",
  "withdrawn"
];

async function seedBorrowers(count = 20) {
  console.log("🌱 Seeding fake borrowers...");

  for (let i = 0; i < count; i++) {
    const passwordHash = await bcrypt.hash("password123", 10);

    const borrower = await db.user.create({
      data: {
        email: faker.internet.email().toLowerCase(),
        passwordHash,
        role: "borrower",
        firstName: faker.person.firstName(),
        lastName: faker.person.lastName(),
        applicationStatus: BORROWER_STAGES[i % BORROWER_STAGES.length],
        createdAt: faker.date.past(),
        isTestUser: true
      }
    });

    // Optional: create fake loan application
    if (borrower.applicationStatus !== "registered") {
      await db.loanApplication.create({
        data: {
          userId: borrower.id,
          loanAmount: faker.number.int({ min: 150000, max: 3000000 }),
          loanType: faker.helpers.arrayElement([
            "bridge_no_rehab",
            "light_rehab",
            "heavy_rehab"
          ]),
          fico: faker.number.int({ min: 660, max: 780 }),
          status: borrower.applicationStatus
        }
      });
    }
  }

  console.log("✅ Seed complete");
}

seedBorrowers()
  .then(() => process.exit(0))
  .catch(err => {
    console.error(err);
    process.exit(1);
  });

5️⃣ Add a script command to run it

In package.json:

{
  "scripts": {
    "seed:borrowers": "node server/scripts/seedBorrowers.js"
  }
}


Run it with:

npm run seed:borrowers

6️⃣ Protect yourself (VERY IMPORTANT)

Never let this run in prod accidentally.

Add a guard:

if (process.env.NODE_ENV === "production") {
  throw new Error("❌ Seeding is disabled in production");
}


Put it at the top of the seed file.

7️⃣ How pros take this further (optional but powerful)

Once this works, teams usually add:

C) Deterministic seeds

This makes bugs reproducible.
