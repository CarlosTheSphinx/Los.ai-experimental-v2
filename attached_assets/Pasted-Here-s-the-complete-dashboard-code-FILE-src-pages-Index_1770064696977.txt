```tsx
import React, { createContext, useContext, useState, ReactNode } from 'react';
import { UserRole } from '@/types/loan';

interface User {
  id: string;
  name: string;
  email: string;
  role: UserRole;
}

interface AuthContextType {
  user: User | null;
  role: UserRole;
  setRole: (role: UserRole) => void;
  isAuthenticated: boolean;
}

const AuthContext = createContext(undefined);

const mockUsers: Record = {
  lender: { id: 'lender-1', name: 'Alex Thompson', email: 'alex@lender.com', role: 'lender' },
  borrower: { id: 'borrower-1', name: 'John Smith', email: 'john.smith@email.com', role: 'borrower' },
  broker: { id: 'broker-1', name: 'Maria Garcia', email: 'maria@abcmortgage.com', role: 'broker' },
};

export function AuthProvider({ children }: { children: ReactNode }) {
  const [role, setRole] = useState('lender');
  const user = mockUsers[role];

  return (

      {children}

  );
}

export function useAuth() {
  const context = useContext(AuthContext);
  if (context === undefined) throw new Error('useAuth must be used within an AuthProvider');
  return context;
}
```
