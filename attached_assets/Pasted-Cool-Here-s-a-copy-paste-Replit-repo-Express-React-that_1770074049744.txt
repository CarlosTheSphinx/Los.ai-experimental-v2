Cool. Here’s a **copy-paste Replit repo** (Express + React) that gives you a working **Quote Maker** + **Admin Ruleset Builder**.

This design does exactly what you described:

* Admin creates a **Product** + **Ruleset version**
* Admin can upload guidelines (PDF) to store/extract text (optional)
* Quote Maker lets user pick a product + ruleset version, input loan scenario, and get **deterministic pricing** from the engine

You can drop this into a new Replit Node.js project and run.

```js
import Database from "better-sqlite3";

const db = new Database("app.db");

db.exec(`
  PRAGMA journal_mode = WAL;

  CREATE TABLE IF NOT EXISTS products (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL,
    category TEXT NOT NULL,
    status TEXT NOT NULL DEFAULT 'active',
    created_at TEXT NOT NULL DEFAULT (datetime('now'))
  );

  CREATE TABLE IF NOT EXISTS rulesets (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    product_id INTEGER NOT NULL,
    version INTEGER NOT NULL,
    rules_json TEXT NOT NULL,
    created_at TEXT NOT NULL DEFAULT (datetime('now')),
    FOREIGN KEY(product_id) REFERENCES products(id)
  );

  CREATE TABLE IF NOT EXISTS uploads (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    product_id INTEGER NOT NULL,
    filename TEXT NOT NULL,
    extracted_text TEXT,
    created_at TEXT NOT NULL DEFAULT (datetime('now')),
    FOREIGN KEY(product_id) REFERENCES products(id)
  );

  CREATE TABLE IF NOT EXISTS quotes (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    product_id INTEGER NOT NULL,
    ruleset_id INTEGER NOT NULL,
    inputs_json TEXT NOT NULL,
    outputs_json TEXT NOT NULL,
    created_at TEXT NOT NULL DEFAULT (datetime('now')),
    FOREIGN KEY(product_id) REFERENCES products(id),
    FOREIGN KEY(ruleset_id) REFERENCES rulesets(id)
  );
`);

export function createProduct({ name, category }) {
  const stmt = db.prepare(`INSERT INTO products (name, category) VALUES (?, ?)`);
  const info = stmt.run(name, category);
  return info.lastInsertRowid;
}

export function listProducts() {
  return db.prepare(`SELECT * FROM products ORDER BY created_at DESC`).all();
}

export function getProduct(id) {
  return db.prepare(`SELECT * FROM products WHERE id = ?`).get(id);
}

export function createRuleset({ productId, rulesJson }) {
  const latest = db
    .prepare(`SELECT MAX(version) as v FROM rulesets WHERE product_id = ?`)
    .get(productId);
  const nextVersion = (latest?.v ?? 0) + 1;

  const stmt = db.prepare(`
    INSERT INTO rulesets (product_id, version, rules_json)
    VALUES (?, ?, ?)
  `);

  const info = stmt.run(productId, nextVersion, JSON.stringify(rulesJson));
  return { rulesetId: info.lastInsertRowid, version: nextVersion };
}

export function listRulesets(productId) {
  return db
    .prepare(`SELECT id, product_id, version, created_at FROM rulesets WHERE product_id = ? ORDER BY version DESC`)
    .all(productId);
}

export function getRuleset(rulesetId) {
  const row = db.prepare(`SELECT * FROM rulesets WHERE id = ?`).get(rulesetId);
  if (!row) return null;
  return { ...row, rules_json: JSON.parse(row.rules_json) };
}

export function saveQuote({ productId, rulesetId, inputs, outputs }) {
  const stmt = db.prepare(`
    INSERT INTO quotes (product_id, ruleset_id, inputs_json, outputs_json)
    VALUES (?, ?, ?, ?)
  `);
  const info = stmt.run(productId, rulesetId, JSON.stringify(inputs), JSON.stringify(outputs));
  return info.lastInsertRowid;
}

export function saveUpload({ productId, filename, extractedText }) {
  const stmt = db.prepare(`
    INSERT INTO uploads (product_id, filename, extracted_text)
    VALUES (?, ?, ?)
  `);
  const info = stmt.run(productId, filename, extractedText ?? null);
  return info.lastInsertRowid;
}

export default db;
```

## `/server/pricing/matcher.js`

```js
// Simple matcher used by ruleset conditions.
// Add more operators as you evolve your products.
export function matches(when, inputs) {
  if (!when) return true;

  // direct equals checks
  for (const key of ["loanType", "purpose", "tier", "propertyType", "state"]) {
    if (when[key] !== undefined && when[key] !== inputs[key]) return false;
  }

  // numeric operators
  if (typeof when.ficoLt === "number" && !(inputs.fico < when.ficoLt)) return false;
  if (typeof when.ficoGte === "number" && !(inputs.fico >= when.ficoGte)) return false;

  // boolean
  if (when.isMidstream === true && inputs.isMidstream !== true) return false;

  // array membership
  if (Array.isArray(when.msaIn)) {
    if (!inputs.msa || !when.msaIn.includes(inputs.msa)) return false;
  }

  return true;
}
```

## `/server/pricing/engine.js`

```js
import { matches } from "./matcher.js";

function clamp(n, min, max) {
  return Math.max(min, Math.min(max, n));
}

export function validateRuleset(rules) {
  const errors = [];
  if (!rules?.product) errors.push("Missing rules.product");
  if (!rules?.baseRates) errors.push("Missing rules.baseRates");
  if (!Array.isArray(rules?.leverageCaps)) errors.push("Missing rules.leverageCaps[]");
  return errors;
}

export function priceQuote(rules, inputs) {
  const errors = validateRuleset(rules);
  if (errors.length) return { eligible: false, reasons: errors };

  const baseRate = rules.baseRates?.[inputs.loanType];
  if (typeof baseRate !== "number") {
    return { eligible: false, reasons: [`Unknown loanType: ${inputs.loanType}`] };
  }

  // Eligibility rules
  const reasons = [];
  for (const r of rules.eligibilityRules ?? []) {
    if (matches(r.when, inputs)) {
      if (r.result === "ineligible") reasons.push(r.label);
    }
  }
  if (reasons.length) return { eligible: false, reasons };

  // Cap selection
  const cap = (rules.leverageCaps ?? []).find(
    (c) => c.tier === inputs.tier && (c.loanTypes ?? []).includes(inputs.loanType)
  );

  let maxLTC = cap?.max?.ltc ?? 0;
  let maxLTAIV = cap?.max?.ltaiv ?? 0;
  let maxLTARV = cap?.max?.ltarv ?? 0;

  // Overlays (cap adjustments)
  const notes = [];
  for (const o of rules.overlays ?? []) {
    if (matches(o.when, inputs)) {
      if (o.effects?.ltcAdd) maxLTC += o.effects.ltcAdd;
      if (o.effects?.ltaivAdd) maxLTAIV += o.effects.ltaivAdd;
      if (o.effects?.ltarvAdd) maxLTARV += o.effects.ltarvAdd;
      notes.push(o.label);
    }
  }

  maxLTC = clamp(maxLTC, 0, 0.95);
  maxLTAIV = clamp(maxLTAIV, 0, 0.95);
  maxLTARV = clamp(maxLTARV, 0, 0.95);

  // Rate + points adjusters
  let rate = baseRate;
  let points = rules.points?.default ?? 0;
  const appliedAdjusters = [];

  for (const a of rules.adjusters ?? []) {
    if (matches(a.when, inputs)) {
      const rateAdd = a.rateAdd ?? 0;
      const pointsAdd = a.pointsAdd ?? 0;
      rate += rateAdd;
      points += pointsAdd;
      appliedAdjusters.push({ id: a.id, label: a.label, rateAdd, pointsAdd });
    }
  }

  return {
    eligible: true,
    baseRate,
    finalRate: Number(rate.toFixed(3)),
    points: Number(points.toFixed(3)),
    caps: { maxLTC, maxLTAIV, maxLTARV },
    appliedAdjusters,
    notes
  };
}
```

```js
import { createRuleset, listRulesets, getRuleset } from "../db.js";

  const out = createRuleset({ productId, rulesJson });
  res.json(out);
```

## `/server/routes/quotes.js`

```js
import { Router } from "express";
import { getRuleset, saveQuote } from "../db.js";
import { priceQuote } from "../pricing/engine.js";

const router = Router();

/**
 * POST /api/quotes/price
 * body: { productId, rulesetId, inputs }
 */
router.post("/price", (req, res) => {
  const { productId, rulesetId, inputs } = req.body ?? {};
  if (!productId || !rulesetId || !inputs) {
    return res.status(400).json({ error: "productId, rulesetId, inputs required" });
  }

  const ruleset = getRuleset(Number(rulesetId));
  if (!ruleset) return res.status(404).json({ error: "ruleset not found" });

  const output = priceQuote(ruleset.rules_json, inputs);

  // Save quote for audit trail
  const quoteId = saveQuote({
    productId: Number(productId),
    rulesetId: Number(rulesetId),
    inputs,
    outputs: output
  });

  res.json({ quoteId, ...output });
});

export default router;
```

## `/server/routes/uploads.js`

```js
import { Router } from "express";
import multer from "multer";
import pdfParse from "pdf-parse";
import { saveUpload } from "../db.js";

const router = Router();
const upload = multer({ storage: multer.memoryStorage() });

/**
 * POST /api/uploads/:productId
 * form-data: file=<pdf>
 * returns extracted text (basic)
 */
router.post("/:productId", upload.single("file"), async (req, res) => {
  const productId = Number(req.params.productId);
  if (!req.file) return res.status(400).json({ error: "file required" });

  let extractedText = "";
  try {
    if (req.file.mimetype === "application/pdf") {
      const parsed = await pdfParse(req.file.buffer);
      extractedText = parsed.text ?? "";
    } else {
      extractedText = req.file.buffer.toString("utf-8");
    }
  } catch (e) {
    return res.status(500).json({ error: "failed to extract text", details: String(e) });
  }

  const uploadId = saveUpload({
    productId,
    filename: req.file.originalname,
    extractedText
  });

  res.json({ uploadId, filename: req.file.originalname, extractedText });
});

export default router;
```

```jsx
import { useMemo, useState } from "react";

export default function JsonEditor({ initialObject, onSave }) {
  const [text, setText] = useState(() => JSON.stringify(initialObject, null, 2));
  const [error, setError] = useState("");

  const parsed = useMemo(() => {
    try {
      const obj = JSON.parse(text);
      setError("");
      return obj;
    } catch (e) {
      setError(String(e));
      return null;
    }
  }, [text]);

  return (
    <div>
      <textarea
        value={text}
        onChange={(e) => setText(e.target.value)}
        style={{
          width: "100%",
          minHeight: 280,
          fontFamily: "ui-monospace, SFMono-Regular, Menlo, Monaco, Consolas",
          fontSize: 12
        }}
      />
      {error && <div style={{ color: "crimson", marginTop: 6 }}>{error}</div>}

      <div style={{ marginTop: 10 }}>
        <button disabled={!parsed} onClick={() => onSave(parsed)}>
          Save Ruleset Version
        </button>
      </div>
    </div>
  );
}
```

```jsx
function UploadBox({ productId }) {
  const [file, setFile] = useState(null);
  const [text, setText] = useState("");

  async function upload() {
    if (!file || !productId) return;

    const fd = new FormData();
    fd.append("file", file);

    const res = await fetch(`/api/uploads/${productId}`, { method: "POST", body: fd });
    const out = await res.json();
    setText(out.extractedText || "");
  }

  return (
    <div>
      <input type="file" accept=".pdf,.txt" onChange={(e) => setFile(e.target.files?.[0] || null)} />
      <div style={{ marginTop: 10 }}>
        <button onClick={upload} disabled={!file || !productId}>
          Upload + Extract
        </button>
      </div>

      {text && (
        <textarea
          value={text}
          readOnly
          style={{
            width: "100%",
            minHeight: 160,
            marginTop: 10,
            fontFamily: "ui-monospace, SFMono-Regular, Menlo, Monaco, Consolas",
            fontSize: 12
          }}
        />
      )}
    </div>
  );
}
```

* auditability
* version control
* deterministic quoting
